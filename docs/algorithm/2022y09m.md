public static boolean[] setPrime(int number){
    boolean[] result = new boolean[number + 1];
    result[0] = true;
    result[1] = true;

    for(int i = 2 ; i <= Math.sqrt(number) ; i++){
        if(result[i]){
            continue;
        }
        for(int j = i * i ; j <= number ; j += i){
            result[j] = true;
        }
    }
    return result;
}
