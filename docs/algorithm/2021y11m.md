# **`[백준] 그래프 BFS` [이모티콘](https://www.acmicpc.net/problem/14226) 진행 중**

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.*;

class Emoticon{
    int emoCount;
    int time;
    int clipboard;
    public Emoticon(int emoCount, int time, int clipboard) {
        this.emoCount = emoCount;
        this.time = time;
        this.clipboard = clipboard;

    }
    @Override
    public String toString() {
        return "Emoticon [emoCount=" + emoCount + ", time=" + time + ", clipboard=" + clipboard + "]";
    }

}

class Main {
    static List<ArrayList<Integer>> list;
    static boolean[][] checked = new boolean[2001][2001];
    static StringBuilder resultString = new StringBuilder();
    static int result = 0;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int target = Integer.parseInt(br.readLine());
        System.out.println(solve(target));
    }

    public static int solve(int target) {
        Queue<Emoticon> queue = new LinkedList<Emoticon>();
        queue.offer(new Emoticon(1 , 0 , 0));
        while(!queue.isEmpty()) {
            Emoticon now = queue.poll();
//            System.out.println(now);
            if(now.emoCount == target) return now.time;
            if(!checked[now.emoCount][now.emoCount]) {
                queue.offer(new Emoticon(now.emoCount , now.time + 1 , now.emoCount));
                checked[now.emoCount][now.emoCount] = true;
                System.out.println(queue.peek());
            }
            if(now.clipboard > 0 && (now.emoCount + now.clipboard) < 2001 && !checked[now.emoCount + now.clipboard][now.emoCount]) {
                queue.offer(new Emoticon(now.emoCount + now.clipboard , now.time + 1 , 0));
                checked[now.emoCount + now.clipboard][now.emoCount] = true;
                System.out.println(queue.peek());
            }
            if(now.emoCount - 1 >= 1 && !checked[now.emoCount - 1][now.emoCount - 1]) {
                queue.offer(new Emoticon(now.emoCount - 1 , now.time + 1 , 0));
                checked[now.emoCount - 1][now.emoCount - 1] = true;
                System.out.println(queue.peek());
            }
        }
        return 0;
    }
}
```
